# 表单

<template>
  <Common-compTable-index
  :tableData="tableData"
  :tableItems="tableItems"
  :buttonItems="buttonItems"
    ></Common-compTable-index>
</template>
<script>
  export default {
    data() {
      return {
         tableData: [{
        name: 890899090987,
        realname: '第一个测试工程',
      },
      {
        name: 89223322387,
        realname: '第二个测试工程',
      }, {
        name: 89089999999,
        realname: '第三个测试工程',
      }],
         buttonItems: {
        label: '操作',
        options: [
          {
            text: '打开',
            cb: 'open',
            type: 'success'
          },
          {
            text: '编辑',
            cb: 'edit',
            type: 'primary'
          },
          {
            text: '删除',
            cb: 'delete',
            type: 'danger'
          }
        ]
      },
      tableItems: [
        { label: '工程名称', name: 'name', sortable: true, formatter: 'datetime' },
        { label: '详细信息', name: 'realname', sortable: true, copy: true }
      ]
      }
    }
  }
</script>
